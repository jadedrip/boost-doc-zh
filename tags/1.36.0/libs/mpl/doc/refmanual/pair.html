pair

摘要

template<
      typename T1
    , typename T2
    >
struct pair
{
    typedef pair type;
    typedef T1 first;
    typedef T2 second;
};

描述

一个透明的保存两个任意类型的模板。

头文件

#include <boost/mpl/pair.hpp>

例子

计算某个序列中的元素数量，并同时计算其中的负数元素数量。

typedef fold<
      vector_c<int,-1,0,5,-7,-2,4,5,7>
    , pair< int_<0>, int_<0> >
    , pair< 
          next< first<_1> >
        , if_< 
              less< _2, int_<0> >
            , next< second<_1> >
            , second<_1> 
            >
        >
    >::type p;

BOOST_MPL_ASSERT_RELATION( p::first::value, ==, 8 );
BOOST_MPL_ASSERT_RELATION( p::second::value, ==, 3 );
